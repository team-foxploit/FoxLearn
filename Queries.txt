CREATE TABLE STUDENT (
Std_Id INT NOT NULL AUTO_INCREMENT,
First_Name VARCHAR(45) NOT NULL,
Last_Name VARCHAR(45) NOT NULL,
Password VARCHAR(10) NOT NULL,
Email VARCHAR(45) NOT NULL,
User_Name VARCHAR(45) NOT NULL,PRIMARY KEY (Std_Id))
ENGINE = InnoDB;


CREATE TABLE TEACHER (
Teacher_Id INT NOT NULL AUTO_INCREMENT,
First_Name VARCHAR(45) NOT NULL,
 Last_Name VARCHAR(45) NOT NULL,
 Password VARCHAR(45) NOT NULL,
 Email VARCHAR(45) NOT NULL,
User_Name VARCHAR(45) NOT NULL,PRIMARY KEY (Teacher_Id))
ENGINE = InnoDB;



CREATE TABLE SUBJECT(
Subject_Id INT NOT NULL AUTO_INCREMENT,Subject_Name VARCHAR(45) NOT NULL,
PRIMARY KEY (Subject_Id))
ENGINE = InnoDB;


CREATE TABLE ANSWERS (Answer_Id INT NOT NULL AUTO_INCREMENT,Answer VARCHAR(45) NOT NULL,Question_Id VARCHAR(255) NOT NULL,  PRIMARY KEY (Answer_Id,Question_Id));

CREATE TABLE QUESTIONS(Question_Id VARCHAR(255) NOT NULL,Correct_Answer_Id INT NOT NULL,Question VARCHAR(500) NOT NULL, Subject_Id INT NOT NULL,Teacher_Id INT NOT NULL,DIFFICULTY VARCHAR(10) NOT NULL,PRIMARY KEY (Question_Id));

CREATE TABLE CHEMISTRY_SCORE (

Atmp_3 DOUBLE NULL,
Atmp_4 DOUBLE NULL,
Atmp_5 DOUBLE NULL,PRIMARY KEY (Std_Id)) 

ENGINE = InnoDB;
